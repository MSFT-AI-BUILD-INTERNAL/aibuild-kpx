```python
def _list_panels(doc_path: Path) -> list[str]:
    text = doc_path.read_text(encoding="utf-8")
    ids = re.findall(r"<!--\s*panel:([A-Za-z0-9_\-]+):start", text)
    seen: dict[str, None] = {}
    for i in ids:
        seen.setdefault(i, None)
    return list(seen)
```
